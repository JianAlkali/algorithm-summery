[[_数据结构]]

前置：并查集（由于带权并查集也可以当并查集用，不再另外开页）。在并查集的基础上记录到头的距离。其的`unite`有方向性。
```cpp
int fa[N], dis[N], sz[N];
void init(int n){
    for(int i=0;i<=n;i++){
        fa[i]=i;
        dis[i]=0;
        sz[i]=1;
    }
}
int find(int i){
    if(fa[i]!=i){
        int rt=find(fa[i]);
        dis[i]+=dis[fa[i]];
        fa[i]=rt;
    }
    return fa[i];
}
bool unite(int i,int j){ // i接到j后面
    int k=find(i), l=find(j);
    if(k==l) return false;
    // if(k<l) swap(k,l); // 这一步可让顺序仅依赖大小
    fa[k]=l; // k->l
    dis[k]=sz[l];
    sz[l]+=sz[k];
    return true;
}
int query(int i,int j){ // i在j后多少位
    int k=find(i), l=find(j);
    if(k!=l) return INF;
    return dis[i]-dis[j];
}
```
