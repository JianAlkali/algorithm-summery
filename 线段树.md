[[数据结构]]

$O\left(logn\right)$的区间操作。感谢lxz提供的线段树板子。此线段树支持操作：区间加、乘、赋值、区间查询。
```cpp
#define ls(x) ((x)<<1)
#define rs(x) ((x)<<1 | 1)
class Seg{ //带lazy，有区间加法、区间乘法、区间赋值和区间查询操作的超级牛（累）的线段树板子，绝大部分参考lxz（除了取模这里没加）
    struct Nd{ //带lazy节点
        int sum,add,mul,st; //lazy操作指的是他的孩子要执行而未应用的操作
        bool hasst;
        Nd()=default;
        Nd(int sum,int add=0,int mul=1,int st=0,bool hasst=false)
            :sum(sum),add(add),mul(mul),st(st),hasst(hasst){}
    }t[N<<2]; //4倍空间
    int n; //这是1based线段树：[1,n]
    void _pushup(int x){ //将更新的sum推上去
        t[x].sum=t[ls(x)].sum+t[rs(x)].sum;
    }
    void _apply(int x,int l,int r,const Nd& upd){ //应用到x
        if(upd.hasst){ //如果有赋值要应用
            t[x].sum=upd.st*(r-l+1); //应用赋值
            //x本身的更新将没用，赋值变为upd的
            t[x].add=0; t[x].mul=1;
            t[x].hasst=true; t[x].st=upd.st;
        }
        //如果有乘法和除法要应用
        t[x].sum=t[x].sum*upd.mul+upd.add*(r-l+1); //先乘后加
        //更新x的lazy，因为对x的更新也会影响x对它孩子的更新
        if(t[x].hasst){ //若x为set操作，lazy的传递类似上面
            //add不用乘(r-l+1)，因为那个是在x作为upd的时候乘的，别double op!
            t[x].st=t[x].st*upd.mul+upd.add;
        }else{ //否则，操作需要叠加
            t[x].add=t[x].add*upd.mul+upd.add;
            t[x].mul=t[x].mul*upd.mul;
        }
    }
```

```cpp
    void _pushdown(int x,int l,int r){ //将lazy操作下推
        if(t[x].add!=0 || t[x].mul!=1 || t[x].hasst){ //有lazy
            //具体将当前节点lazy应用到孩子
            int m=(l+r)>>1;
            _apply(ls(x),l,m,t[x]);
            _apply(rs(x),m+1,r,t[x]);
            //清空x的lazy标记
            t[x].add=0;
            t[x].mul=1;
            t[x].hasst=false;
            t[x].st=0;
        }
    }
    void _build(int x,int l,int r,const array<int,N>& a){
        if(l==r){
            t[x]=Nd(a[l]); //叶子节点
            return;
        }
        t[x]=Nd();
        int m=(l+r)>>1;
        _build(ls(x),l,m,a);
        _build(rs(x),m+1,r,a);
        _pushup(x); //上推sum
    }
    void _update(int x,int l,int r,int ql,int qr,const Nd& upd){
        if(qr<l || r<ql)
            return;
        if(ql<=l && r<=qr){ //当前区间完全覆盖
            _apply(x,l,r,upd); //直接应用更新
            return;
        }
        _pushdown(x,l,r); //先将待处理的lazy下传，否则大问题
        int m=(l+r)>>1;
        if(ql<=m) _update(ls(x),l,m,ql,qr,upd);
        if(qr>m) _update(rs(x),m+1,r,ql,qr,upd);
        _pushup(x); //将修改的孩子上推
}
```

```cpp
    int _query(int x,int l,int r,int ql,int qr){
        if(qr<l || r<ql)
            return 0;
        if(ql<=l && r<=qr){
            return t[x].sum;
        }
        _pushdown(x,l,r);
        int res=0;
        int m=(l+r)>>1;
        if(ql<=m) res+=_query(ls(x),l,m,ql,qr);
        if(qr>m) res+=_query(rs(x),m+1,r,ql,qr);
        return res;
    }
public:
    Seg(int n):n(n){
        _build(1,1,n,array<int,N>{}); //不传入array，默认构造节点初始都是0
    }
    Seg(const array<int,N>& a,int n):n(n){ //带初始节点值的构造
        _build(1,1,n,a);
    }
    void add(int l,int r,int val){
        _update(1,1,n,l,r,Nd(0,val,1,0,false));
    }
    void mul(int l,int r,int val){
        _update(1,1,n,l,r,Nd(0,0,val,0,false));
    }
    void set(int l,int r,int val){ //区间赋值
        _update(1,1,n,l,r,Nd(0,0,1,val,true));
    }
    int query(int l,int r){ //区间查询
        return _query(1,1,n,l,r);
    }
};
```
