[[_数据结构]]

其时间、空间复杂度与线段树一致，但常数更小。但不支持懒标记有关操作。一般用于`(1)`单点修改+区间查询和`(2)`区间修改+单点查询，`(2)`是`(1)`加上拆分的思想，两种类型的修改和查询均为$O\left(\log n\right)$。下面两个分别是`(1)`和`(2)`类型的树状数组：
## (1) 单点修改+区间查询
```cpp
#define low(x) ((-(x))&(x))
struct Fen{ // 1base
    int a[N];
    int n;
    inl void add(int p,int v){
        while(p<=n){
            a[p]+=v; // 若记录前缀最大值，改成a[p]=max(a[p],v)
            p+=low(p);
        }
    }
    inl int query(int r){ //[1,r]
        int res=0; // 若记录前缀最大值，改成=-INF
        while(r>=1){
            res+=a[r]; // 若记录前缀最大值，改成res=max(res,a[p])
            r-=low(r);
        }
        return res;
    }
    inl int query(int l,int r){ //[l,r]
        return query(r)-query(l-1);
    }
    inl void rebuild(int _n){
        n=_n;
        fill(a,a+n+1,0); // 若记录前缀最大值，改成-INF
    }
    inl void rebuild(int v[N],int _n){ // 带初始值的rebuild
        rebuild(_n);
        for(int i=1;i<=n;i++)
            add(i,v[i]);
    }
}fen;
```
## (2) 区间修改+单点查询
```cpp
#define low(x) ((-(x))&(x))
struct Fen{ // 1base
    int a[N]; // 记录拆分，这里取名d[N]更符合含义
    int n;
    inl void _add(int p,int v){ //同(1)
        while(p<=n){
            a[p]+=v;
            p+=low(p);
        }
    }
    inl void add(int r,int v){ //[1,r]
        _add(1,v);
        if(r!=n) _add(r+1,-v);
    }
    inl void add(int l,int r,int v){ //[l,r]
        _add(l,v);
        if(r!=n) _add(r+1,-v);
    }
    inl int query(int p){
        int res=0;
        while(p>=1){
            res+=a[p];
            p-=low(p);
        }
        return res;
    }
    inl void rebuild(int _n){
        n=_n;
        fill(a,a+n+1,0);
    }
    inl void rebuild(int v[N],int _n){
        rebuild(_n);
        if(n) _add(1,v[1]);
        for(int i=2;i<=n;i++)
            _add(i,v[i]-v[i-1]);
    }
}fen;
```
