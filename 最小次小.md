[[其他技巧&我的小东西]]

许多情况需要在统计最大/小的同时统计次大/小，我们不妨将add操作抽象成一个模版来实现这一过程。模版均为最小次小，并可无缝替换为最大次大，同时多数可以扩展到第k小统计，通常拥有$O\left(k\right)$的复杂度（k较大时不建议）。
朴素最小次小：
```cpp
struct M2{ //m1最小m2次小
    int m1=INF, m2=INF;
    inl int add(int val){
        if(val<m1){
            m2=m1;
            m1=val;
            return 1; //upd 1
        }else if(val<m2){
            m2=val;
            return 2; //upd 2
        }
        return 0; //no upd
    }
};
```

朴素最小次小k版本：（k较小时确实比set快）
```cpp
constexpr int MSZ=3;
struct Mk{
    int mi[MSZ];
    inl int add(int val){
        for(int i=0;i<MSZ;i++){
            if(val<mi[i]){
                for(int j=MSZ-1;j>i;j--)
                    mi[j]=mi[j-1];
                mi[i]=val;
                return i+1;
            }
        }
        return 0;
    }
};
```

统计计两个key不同的最小次小val：
```cpp
struct M2{ //v1最小v2次小
    int k1=-1, k2=-1, v1=INF, v2=INF;
    inl int add(int key, int val){
        if(key==k1){
            if(val<v1){
                v1=val;
                return 1; //upd k1
            }
        }else if(key==k2){
            if(val<v2){
                v2=val;
                return 2; //upd k2
            }
        }else{
            if(val<v1){
                v2=v1; k2=k1;
                v1=val; k1=key;
                return 1;
            }else if(val<v2){
                v2=val; k2=key;
                return 2;
            }
        }
        return 0;
    }
};
```
