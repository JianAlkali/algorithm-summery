[[数据结构]]

通过动态开点，来实现某些操作的线段树。适用于$n$很大但同时无法用离散化的情况。操作的复杂度是$O\left(logn⋅q\right)$同线段树，但空间复杂度降至$O\left(logn⋅q\right)$。
```cpp
struct Seg{
    struct Nd{
        i128 sum,add; // 区间加法、区间查询动态开点线段树
        int ls,rs;
    } t[N*32];
    int n;
    int idx=0, rt=0;
    void init(int _n){
        idx=0; rt=0;
        n=_n;
    }
    void nw(int& x){ // 若点不存在则开新点
        if(!x) x=++idx;
    }
    void apply(int x,int l,int r,const Nd& upd){
        if(upd.add){
            t[x].add+=upd.add;
            t[x].sum+=upd.add*(r-l+1);
        }
    }
    void pushdown(int x,int l,int r){ // 多一步先new
        if(t[x].add){
            nw(t[x].ls);
            nw(t[x].rs);
            int m=(l+r)>>1;
            apply(t[x].ls,l,m,t[x]);
            apply(t[x].rs,m+1,r,t[x]);
            t[x].add=0;
        }
    }
    void pushup(int x){
        t[x].sum=t[t[x].ls].sum+t[t[x].rs].sum;
    }
```

```cpp
// 注意加引用，以更新父节点的孩子信息
void set(int& x,int l,int r,int ql,int qr,const Nd& upd){
        nw(x); // 需要保证x存在
        if(ql<=l && r<=qr){
            apply(x,l,r,upd);
            return;
        }
        int m=(l+r)>>1;
        pushdown(x,l,r);
        if(ql<=m) set(t[x].ls,l,m,ql,qr,upd);
        if(qr>m) set(t[x].rs,m+1,r,ql,qr,upd);
        pushup(x);
    }
    void add(int l,int r,int k){ // 外部接口
        set(rt,1,n,l,r,Nd{0,k,0,0});
    }
    i128 query(int x,int l,int r,int ql,int qr){
        if(!x) return 0; // x不存在则结果就为0，无需开点和递归
        if(ql<=l && r<=qr)
            return t[x].sum;
        int m=(l+r)>>1;
        i128 res=0;
        pushdown(x,l,r);
        if(ql<=m) res+=query(t[x].ls,l,m,ql,qr);
        if(qr>m) res+=query(t[x].rs,m+1,r,ql,qr);
        return res;
    }
    i128 query(int l,int r){ // 外部接口
        return query(rt,1,n,l,r);
    }
} seg;
```
