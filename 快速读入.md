[[_杂项]]

这是快速读入模版（整数）
在一些卡常的时候说不定会有特殊效果。
```cpp
#define getchar getchar_unlocked // 在Linux(大部分OJ)等地可能有效
// #define getchar _getchar_nolock // 在Windows等地可能有效
ll read(){
    ll x=0,sgn=1;
    char c=getchar();
    while(!isdigit(c)){
        if(c=='-') sgn=-1;
        c=getchar();
    }
    while(isdigit(c)){
        x=x*10+(c-'0');
        c=getchar();
    }
    return x*sgn;
}
void write(ll a){
    if(a<0) putchar('-'), a=-a;
    if(a<10) putchar(a+'0');
    else write(a/10), putchar((a%10)+'0');
}
```
