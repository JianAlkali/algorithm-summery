[[_杂项]]

这是快速读入模版（整数）
在一些卡常的时候说不定会有特殊效果。
## 普通快读
```cpp
#define getchar getchar_unlocked // 在Linux(大部分OJ)等地可能有效
// #define getchar _getchar_nolock // 在Windows等地可能有效
ll read(){
    ll x=0,sgn=1;
    char c=getchar();
    while(!isdigit(c)){
        if(c=='-') sgn=-1;
        c=getchar();
    }
    while(isdigit(c)){
        x=x*10+(c-'0');
        c=getchar();
    }
    return x*sgn;
}
void write(ll a){
    if(a<0) putchar('-'), a=-a;
    if(a<10) putchar(a+'0');
    else write(a/10), putchar((a%10)+'0');
}
```

## 极限快读
```cpp
namespace FastIO {
    static const int S = 1 << 20;

    char ibuf[S], *iS = ibuf, *iT = ibuf;
    char obuf[S], *oS = obuf;

    // ======== 输入 ========

    inline char gc() {
        if (iS == iT) {
            iT = (iS = ibuf) + fread(ibuf, 1, S, stdin);
            if (iS == iT) return EOF;
        }
        return *iS++;
    }

	// 读入整数，调用示例：int n=FastIO::read<int>();
    template<typename T>
    inline T read() {
        T x = 0, f = 1;
        char c = gc();

        // 跳非数字（含负号）
        while (c < '0' || c > '9') {
            if (c == '-') f = -1;
            c = gc();
        }

        // 读数字
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c & 15);
            c = gc();
        }

        return x * f;
    }
    
    inline string read_line() {
        string s;
        char c = gc();

        // 跳过可能的 '\n' 或 '\r'
        while (c == '\n' || c == '\r') c = gc();

        while (c != '\n' && c != '\r' && c != EOF) {
            s.push_back(c);
            c = gc();
        }

        return s;
    }

    // ======== 输出 ========

    inline void pc(char c) {
        if (oS - obuf == S) {
            fwrite(obuf, 1, S, stdout);
            oS = obuf;
        }
        *oS++ = c;
    }

    template<typename T>
    inline void write(T x) {
        if (x == 0) {
            pc('0');
            return;
        }
        if (x < 0) {
            pc('-');
            x = -x;
        }

        char s[24];
        int t = 0;
        while (x) {
            s[t++] = (x % 10) + '0';
            x /= 10;
        }
        while (t--) pc(s[t]);
    }

    // 在退出程序前调用！
    inline void flush() {
        fwrite(obuf, 1, oS - obuf, stdout);
    }
}
```
