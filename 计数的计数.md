维护计数的数据结构。$O(1$)修改、查询单个元素的`cnt`、其主要目的是$O\left(1\right)$查询`cnt`为某个值的元素有几个、哪几个。此处`cal_dis`会返回需要修改的大小为`dissz`的`dis`。
```cpp
vi c2n[N];
int pp[N];
int cnt[N];    
int tot=0;
    auto add=[&](int p){
        tot++;
        cal_dis(p);
        for(int i=0;i<dsz;i++){
            int w=dis[i], c=cnt[w]++;
            if(c!=0){
                pp[c2n[c].back()]=pp[w];
                swap(c2n[c][pp[w]],c2n[c].back());
                c2n[c].pop_back();
            }
            pp[w]=c2n[c+1].size();
            c2n[c+1].emplace_back(w);
        }
    };
    auto sub=[&](int p){
        tot--;
        cal_dis(p);
        for(int i=0;i<dsz;i++){
            int w=dis[i], c=cnt[w]--;
            pp[c2n[c].back()]=pp[w];
            swap(c2n[c][pp[w]],c2n[c].back());
            c2n[c].pop_back();
            if(c-1!=0){
                pp[w]=c2n[c-1].size();
                c2n[c-1].emplace_back(w);
            }
        }
    };
```
