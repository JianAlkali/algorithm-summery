[[数论]]

$O\left(nlogn\right)$计算平面最近点对，利用鸽巢原理来避免最坏$O\left(n^{2}\right)$的情况：在中间带（宽度为$\sqrt{nowbst}$即最小x距离内，每个点最多只需要与后续6个点进行比较。这是因为：
中间带被分成若干个高度为$\frac{\sqrt{nowbst}}{2}$的小矩形
每个小矩形内最多包含1个点（否则距离会小于当前最小距离）
对于任意点，只需要检查其后续的有限个点
```cpp
pii a[N], tmp[N];
void solve(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i].first>>a[i].second;
    sort(a,a+n);
    auto disqrx=[](pii& a,pii& b)->int { // x距离的平方
        return (a.first-b.first)*(a.first-b.first);
    };
    auto disqr=[](pii& a,pii& b)->int { // (x,y)欧式距离的平方
        return (a.first-b.first)*(a.first-b.first) + (a.second-b.second)*(a.second-b.second);
    };
    double ans=INF;
```

```cpp
    auto merge=[&](auto&& merge,int l,int r)->double {
        int nowbst=INF;
        if(r-l<3){
            for(int i=l;i<=r;i++){
                for(int j=l;j<=r;j++){
                    if(i==j) continue;
                    nowbst=min(nowbst,disqr(a[i],a[j]));
                    ans=min(ans,sqrt(nowbst));
                }
            }
            return nowbst;
        }else{
            int m=(l+r)>>1;
            nowbst=min(merge(merge,l,m),merge(merge,m+1,r));
            int ql=m, qr=m+1;
            while(ql>l && disqrx(a[ql-1],a[m])<=nowbst) ql--; // 仅这里仅看x
            while(qr<r && disqrx(a[qr+1],a[m+1])<=nowbst) qr++;
            copy(a+ql, a+qr+1, tmp+ql); // [l,r]
            sort(tmp+ql, tmp+qr+1, [](const pii& l, const pii& r){ // 按y排序
                return l.second!=r.second ? l.second<r.second : l.first<r.first;
            });
            for(int i=ql;i<=qr;i++){
                for(int j=i+1;j<=qr;j++){
                    int now=disqr(tmp[i],tmp[j]);
                    if(now>nowbst) break; // 最多循环6次
                    ans=min(ans,sqrt(now));
                }
            }
            return nowbst;
        }
    };
    merge(merge,0,n-1);
    cout<<fixed<<setprecision(4);
    cout<<ans<<'\n';
}
```
