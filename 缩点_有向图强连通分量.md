[[图论]]

这里的缩点是对有向图而言的；若两个点彼此可达，则称这两个点在一个强连通分量（SCC）内，而缩点则是将每个SCC看成一个点来求解某些问题的一种步骤。
输入$n$、$m$代表顶点数和边数，下一行$n$个数$a_{i}$，代表每个点点权，接下来$m$行每行2个数$u$、$v$代表一条$u→v$的有向边。求一条路径使经过的点点权和最大，可以重复走走过的点和边，但每个点点权只能被算一次。很显然相互可达→最优解一定是都选。
```cpp
struct E{
    int v,nxt;
}e[N<<1],se[N<<1]; // s 开头的为SCC的点和边的数据
int head[N],shead[N],ec=0,sec=0;
void add(int u,int v){
    e[++ec]={v,head[u]};
    head[u]=ec;
}
void sadd(int u,int v){
    se[++sec]={v,shead[u]};
    shead[u]=sec;
}
int val[N]; // 原始价值
int dfn[N],low[N],dfnt=0; // tarjan所用dfs序、最小回溯、计数
int instk[N],stk[N],top=0; // tarjan所用手工栈和入栈标记
int sval[N],sid[N],in[N],scnt=0; // SCC的数据，该scc价值和，原始点对应SCC点编号、SCC点入度
int dp[N]; // 拓扑排序求最大点权
void solve(){
    int n,m;
    cin>>n>>m;
    ec=sec=dfnt=top=scnt=0;
    for(int i=0;i<=n;i++){
        head[i]=shead[i]=dfn[i]=low[i]=instk[i]=sval[i]=in[i]=dp[i]=0;
    }
    for(int i=1;i<=n;i++)
        cin>>val[i];
    for(int i=0,u,v;i<m;i++){
        cin>>u>>v;
        add(u,v);
    }
```

```cpp
    // 1 tarjan求强连通分量
    auto tarj=[&](auto&& tarj,int u)->void {
        dfn[u]=low[u]=++dfnt;
        stk[++top]=u; instk[u]=true;
        for(int i=head[u];i;i=e[i].nxt){
            int v=e[i].v;
            if(!dfn[v]){ // 没vis过
                tarj(tarj,v);
                low[u]=min(low[u],low[v]);
            }else if(instk[v]) // vis过且在栈上，更新low
                low[u]=min(low[u],dfn[v]);
        }
        if(low[u]==dfn[u]){ // 找到一个SCC
            ++scnt;
            while(true){
                int v=stk[top--];
                sid[v]=scnt; // v对应的SCCid
                sval[scnt]+=val[v]; // SCC算上v的价值
                instk[v]=false;
                if(v==u) break;
            }
        }
    };
    for(int i=1;i<=n;i++)
        if(!dfn[i]) tarj(tarj,i); // 图不一定连通
    // 2 将所求SCC数据由原始点映射到SCC点
    for(int u=1;u<=n;u++){
        for(int i=head[u];i;i=e[i].nxt){
            int v=e[i].v;
            int su=sid[u],sv=sid[v];
            if(su==sv) continue; // 不要自环
            sadd(su,sv); // 可能有重边，但不影响拓扑排序
            in[sv]++;
        }
    }
```

```cpp
    // 3 拓扑排序求最大点权
    queue<int> q;
    for(int i=1;i<=scnt;i++){
        dp[i]=sval[i];
        if(in[i]==0) q.emplace(i);
    }
    while(!q.empty()){
        int u=q.front(); q.pop();
        for(int i=shead[u];i;i=se[i].nxt){
            int v=se[i].v;
            dp[v]=max(dp[v],dp[u]+sval[v]);
            if(--in[v]==0){
                q.emplace(v);
            }
        }
    }
    int ans=0;
    for(int i=1;i<=scnt;i++)
        ans=max(ans,dp[i]);
    cout<<ans<<'\n';
}
```
