[[图论]]

是对无向图而言的，边双联通分量内删掉任意一边后图仍连通。对于一个n个节点m条无向边的图，请输出其边双连通分量的个数，并且输出每个边双连通分量。输入第一行，两个整数n和m。接下来m行，每行两个整数 ，表示一条无向边。不保证图为简单图，图中可能有重边和自环。输出第一行一个整数x表示边双连通分量的个数。接下来的x行，每行第一个数a表示该分量结点个数，然后a个数，描述一个边双连通分量。你可以以任意顺序输出边双连通分量与边双连通分量内的结点。

int head[N];

    edge[ecnt]={v,false,head[u]};
    head[u]=ecnt++;
}
bool vis[N];
int dict[N], low[N]; // dfs序和回溯到的最小dfs序
vi ans[N];
int ansn=0;
inline void solve(){
    int n,m;
    cin>>n>>m;
    fill(head,head+n+1,0);
    fill(vis,vis+n+1,false);
    fill(dict,dict+n+1,INF);
    fill(low,low+n+1,INF);
    for(int i=0;i<=n;i++)
        ans[i].clear();
    ansn=0;
    ecnt=2;

        cin>>u>>v;

    }

    int tt=0;
    auto dfs=[&](auto&& dfs,int u,int f)->void{ // 第一次dfs
        vis[u]=true; dict[u]=low[u]=tt++;
        for(int eg=head[u];eg;eg=edge[eg].nxt){
            int v=edge[eg].v;
            if(v==f) continue;
            if(!vis[v]){
                dfs(dfs,v,u);
                low[u]=min(low[u],low[v]);
                if(low[v]>dict[u]) // 标记是桥，也可直接删
                    edge[eg].isb=edge[inv(eg)].isb=true; 
            }else
                low[u]=min(low[u],dict[v]);
        }
    };
    for(int i=1;i<=n;i++) if(!vis[i]){ // 图不一定连通
            tt=0; dfs(dfs,i,0);
    }
    fill(vis,vis+n+1,false); // 第二次dfs也要
    auto edfs=[&](auto&& edfs,int u)->void{ // 第二次dfs
        vis[u]=true;
        ans[ansn].emplace_back(u);
        for(int eg=head[u];eg;eg=edge[eg].nxt){
            int v=edge[eg].v;
            if(!vis[v] && !edge[eg].isb){ // 连通分量内的点收集
                edfs(edfs,v);
            }
        }
    };
    for(int i=1;i<=n;i++)
        if(!vis[i]){
            edfs(edfs,i); ansn++;
        }
    cout<<ansn<<'\n';
    for(int i=0;i<ansn;i++){
        cout<<ans[i].size()<<' ';
        for(int j=0;j<ans[i].size();j++)
            cout<<ans[i][j]<<" \n"[j==ans.size()-1];
    }
}
