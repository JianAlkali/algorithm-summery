[[动态规划]]

n个物品，容量为c，每个物品占用体积v，价值w，最多装d个
二进制优化$O\left(nclogd\right)$
```cpp
for(int i=0;i<n;i++){
    auto [v,w,d]=a[i];
    int sz=0;
    for(int j=1,isloop=1;isloop;j<<=1){
        if(j>=d){
            j=d; isloop=0; //最后一次循环为j=d
        }else
            d-=j;
        int nv=v*j,nw=w*j;
        for(int k=c;k>=nv;k--){
            dp[k]=max(dp[k],dp[k-nv]+nw);
        }
    }
}
```

单调队列优化，常数会更大$O\left(nc\right)$
```cpp
for(int i=0;i<n;i++){ //array队列+除法替换
    auto [v,w,d]=a[i]; //volume w cnt
    copy(all(dp,c+3),tmp.begin());
    for(int j=0;j<v;j++){ //每个段都是独立的。这样，dp[0~c]都只会被访问一次
        qf=0;qt=-1; //维护位置。保证头部始终是最优的值
        for(int k=j,p=0;k<=c;k+=v,p++){
            while(qf<=qt && p-q[qf] > d) //选择物品太多
                qf++; //assert 0<=qf<N
            if(qf<=qt) //更新
                dp[k]=max(tmp[k],tmp[j+v*q[qf]]+(p-q[qf])*w);
            while(qf<=qt && tmp[j+v*q[qt]]+(p-q[qt])*w<=tmp[k]) //在尾部插入新值时，把更不好的淘汰
                qt--; //assert -1<=qt<N
            q[++qt]=p;
        }
    }
}
```
