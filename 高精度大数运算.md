设位数为N、M，不注明时，双目操作N为两数中N较大的那个。
构造：$从输入流/Big/Big&/string/int/vector：O\left(N\right)$；$Big&&/swap├(Big&┤)：O(1)$
转换：$到输出流/string/int：O\left(N\right)$；$to_binary：O\left(N^{2}\right)$
比较：$O\left(min\left(N,M\right)\right)$，平均$O\left(1\right)$
运算：$高 +/-/×/÷/$；$高 +/- 高：O\left(N\right)$；$高 ×/÷/$。
位运算：按位移位：$O\left(Nb\right)$；按10进制移位/pow10：$O\left(N\right)$
求多项式逆（结果是多项式而非大数，不能再进行其他运算）：$O\left(NlogN\right)$
幂：$高^{高}：O\left(M^{2}+NlogNlogM\right)$；$高^{低}：O\left(NlogNlogm\right)$
开方开根（常数大，数过小时有bug）：$O\left(Nlog^{2}N\right)$
gcd lcm：$O\left(NlogN\right)$
同时，多数情况下自加自乘等会比直接加乘快。

        int n=sz==-1?a.size():sz;
        for(int i=1,j=0,bit;i<n;i++){
            bit=n>>1;
            while(j&bit){
                j^=bit; bit>>=1;
            }
            j^=bit;

                std::swap(a[i],a[j]);
        }

            int l2=len>>1;

            //x_{i+1}=x_i-f(x)/f'(x)带入得x_{i+1}=x_i*(2*a*x_i)或=2x_i+x_i*x_i*a（两种效率差不多）

                int sz=x.a.size();
                if(min(sz,n-sz)<=NEWTON_LEVEL){ //位数过小将停止迭代，直接算出
                    return Big(1).set_pow10(n).divmod(x).first;
                }
                int k=(n-sz+2)>>1; //递归规模减半
                int k2=max(0ll,sz-k); //位移量
                Big x2=x; // 截取x的高位部分
                if(k2>0) x2.self_move(-k2);
                int n2=k+x2.a.size(); //下一步求量
                Big y=newton_inv(newton_inv,x2,n2);
                // 牛顿迭代： return y_{i+1} = 2*y_i-x*y_i^2
                return (y<<1).self_move(n-n2-k2) - (x*sqr(y)).self_move(-2*(n2+k2)+n) -1; //y*2-b*y^2-1，-1是保证得出结果微小于正确结果
            };

                x2=b; k2=0;
            }

                res+=1;
                rem-=b;
            }
        }

    inl Big friend gcd(Big a,Big b){ // Onlogn
        if(a==0 || b==0) return 0;
        Big r=1;
        while(true){ // 每两次循环，a和b较大的至少缩减一半，以保证log级别
            if(a.is_odd() && b.is_odd()){
                switch(a.compare(b)){ //大减小
                    case 1: //a>b
                        a-=b;
                        break;
                    case 0:
                        r*=a;
                        return r;
                    case -1:
                        b-=a;
                        break;
                }

            }else{ // a.even && b.even

                r<<=1;
            }
        }
