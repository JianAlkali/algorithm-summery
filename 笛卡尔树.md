[[_数据结构]]

是二叉树，其满足两种性质：中序遍历可得到原序列、每个节点大于（小于）其子节点。构建$O\left(n\right)$。用其可以搞一些有关元素相对大小的静态区间的事。
```cpp
struct Car{
    pii son[N];
    int fa[N];
    int val[N];
    stack<int> stk;
    inl int rebuild(int n,int a[N]){ // 1base 小根堆
        fill(son,son+n+1,pii{0,0});
        fill(fa,fa+n+1,0);
        copy(a+1,a+n+1,val+1); //copy for use
        for(int i=1;i<=n;i++){
            int last=0;
            while(!stk.empty() && a[stk.top()]>a[i]){
                if(last){
                    son[stk.top()].second=last;
                    fa[last]=stk.top();
                }
                last=stk.top();
                stk.pop();
            }
            if(last){
                son[i].first=last;
                fa[last]=i;
            }
            stk.emplace(i);
        }
        int last=0;
        while(!stk.empty()){
            if(last){
                son[stk.top()].second=last;
                fa[last]=stk.top();
            }
            last=stk.top();
            stk.pop();
        }
        return last; //root
    }

inl void in_order(int x){ // 树上中序遍历==数组正序遍历
        if(son[x].first) in_order(son[x].first);
        // 在此进行你想要的处理..
        cout<<x<<"->"<<son[x].first<<' '<<son[x].second<<'\n';
        if(son[x].second) in_order(son[x].second);
    }
} car;
```
