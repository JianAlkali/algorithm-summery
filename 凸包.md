[[_动态规划]]

集中讨论各种凸包的维护
使用`map`，当`key`+++,一定`val`---，`val`大的优：（另一种方法是使用树状数组）
```cpp
    auto insert=[&](int k,int v){ // k+++ v--- and maxize v
        if(mp.empty()){
            mp.emplace(k,v);
            return;
        }
        auto it=mp.lower_bound(k);
        if(it!=mp.end() && it->second>=v) return;
        if(it==mp.end() || it->first!=k) --it;
        while(it->second<=v){
            it=mp.erase(it);
            if(mp.empty() || it==mp.begin()) break;
            --it;
        }
        mp.emplace(k,v);
    };
```
