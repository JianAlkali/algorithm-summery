[[数据结构]]

时间复杂度与线段树一致（建树$O\left(nlogn\right)$，更新$O\left(logn\right)$，查询$O\left(logn\right)$），空间复杂度$O\left(n+qlogn\right)$（q为修改次数）。用于求解静态区间第k小（这时空间$O\left(nlogn\right)$）、需要回溯的线段树的场景（如可持久化并查集）等问题。但它不支持懒标记。所以区间修改操作会变得复杂（需标记永久化）或不可行（同时有区间赋值区间加等）。
普通版本，单点赋值、区间查询：
```cpp
#define mid(l,r) (((l)+(r))>>1)
struct Seg{
    struct Nd{
        int l,r,val;
    }t[N*24]; //节点
    int tsz, n; //节点数、数组大小
    int head[N<<1], vcnt=0; //版本头、版本数
    inl int clone(int x){ // push一个和x一样的
        t[tsz]=t[x]; return tsz++;
    }
    // 本题无需apply pushup pushdown，如果需要，写法同线段树，注意不支持懒标记
    inl void pushup(int x){
        t[x].val=t[t[x].l].val+t[t[x].r].val;
    }
    inl int rebuild(int l,int r,int a[N]){ // ->newid
        int x=tsz++;
        if(l==r){
            t[x]={0,0,a[l]};
            return x;
        }
        int m=mid(l,r);
        t[x].l=rebuild(l,m,a);
        t[x].r=rebuild(m+1,r,a);
        // pushup(x)
        return x;
    }
    inl void rebuild(int a[N],int _n){ // 接口，多测重建
        n=_n;
        vcnt=1;  tsz=1;
        head[0]=rebuild(1,n,a);
    }
```

```cpp
    inl int set(int x,int l,int r,int ql,int qr,int val){ // ->newid
        x=clone(x);
        if(ql<=l && r<=qr){
            t[x].val=val; // apply
            return x;
        }
        // pushdown(x,l,r,upd)
        int m=mid(l,r);
        if(m>=ql)
            t[x].l=set(t[x].l,l,m,ql,qr,val);
        if(m<qr)
            t[x].r=set(t[x].r,m+1,r,ql,qr,val);
        // pushup(x)
        return x;
    }
    inl void set(int pos,int val, int verfrom){ // 接口
        head[vcnt++]=set(head[verfrom],1,n,pos,pos,val);
    }
    inl int query(int x,int l,int r,int ql,int qr){
        if(ql<=l && r<=qr){
            return t[x].val;
        }
        // pushdown(x,l,r,upd)
        int m=mid(l,r);
        int res=0;
        if(m>=ql)
            res+=query(t[x].l,l,m,ql,qr);
        if(m<qr)
            res+=query(t[x].r,m+1,r,ql,qr);
        return res;
    }
    inl int query(int pos, int verfrom){ // 接口
        head[vcnt++]=clone(head[verfrom]); // 本题要求查询也生成新版本
        return query(head[verfrom],1,n,pos,pos);
    }
}seg;
```

    }

用其求解静态区间第k小，构造$O\left(nlogn\right)$，查询$O\left(logn\right)$：
```cpp
#define mid(l,r) (((l)+(r))>>1)
struct Seg{
    struct Nd{
        int l,r,val;
    }t[N*20];
    int tsz=0, n;
    int head[N], vcnt=0;
    inl int clone(int x){
        t[tsz]=t[x];
        return tsz++;
    }
    inl void pushup(int x){
        t[x].val=t[t[x].l].val+t[t[x].r].val;
    }
```

```cpp
    inl int add(int x,int l,int r,int pos){ // 这不是接口别调
        x=clone(x);
        if(l==pos && r==pos){
            t[x].val++; return x;
        }
        int m=mid(l,r);
        if(pos<=m) t[x].l=add(t[x].l,l,m,pos);
        else t[x].r=add(t[x].r,m+1,r,pos);
        pushup(x);
        return x;
    }
    inl int rebuild(int l,int r){ // 建初始空树
        int x=tsz++;
        if(l==r){
            t[x]={0,0,0}; return x;
        }
        int m=mid(l,r);
        t[x].l=rebuild(l,m);
        t[x].r=rebuild(m+1,r);
        return x;
    }
    inl void rebuild(int a[N], int _n){ // 接口，需先离散化a
        tsz=0; vcnt=0; n=_n;
        head[vcnt++]=rebuild(1,n); //树0：空
        for(int i=1;i<=n;i++){ // 第i颗树存储了前a[1~i]的信息
            head[vcnt]=add(head[vcnt-1],1,n,a[i]); vcnt++;//将离散化后的a[i]当做pos插入
        }
    }
    inl int query(int q,int p,int l,int r,int k){ //通过p树-q树得到差
        if(l==r) return l;
        int cnt=t[t[p].l].val-t[t[q].l].val; // 左树元素个数
        int m=mid(l,r);
        if(cnt>=k) //左子树元素个数>=k，第k小在左子树
            return query(t[q].l,t[p].l,l,m,k);
        else //否则
            return query(t[q].r,t[p].r,m+1,r,k-cnt);
    }
    inl int query(int l,int r,int k){ // 接口，查第[l,r]第k小
        return query(head[l-1],head[r],1,n,k);
    }
}seg;
```

调用示例：（luogu P3834 输入：n m；一行n个整数为a[i]初值；m行每行一个询问 [l r k]，对每个询问输出区间[l,r]上第k小的值）
```cpp
int a[N],li[N];
inl void solve(){
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    copy(a+1,a+n+1,li+1); //1base离散化
    sort(li+1,li+n+1);
    int lisz=unique(li+1,li+n+1)-li -1;
    for(int i=1;i<=n;i++)
        a[i]=lower_bound(li+1,li+lisz+1,a[i])-li;
    seg.rebuild(a,n);
    // 我们查询的时候，返回的是离散化后的结果，用其索引li即可得到答案
    for(int i=1,l,r,k;i<=m;i++){
        cin>>l>>r>>k;
        int res=seg.query(l,r,k);
        res=li[res];
        cout<<res<<'\n';
    }
}
```
