```cpp
constexpr double EPS=1e-9;
inline void solve(){
    int n;
    cin>>n;
    vector<vector<double>> mat(n,vector<double>(n+1));
    for(int i=0;i<n;i++)
        for(int j=0;j<=n;j++)
            cin>>mat[i][j]; 
    //高斯消元。目标：转换方程系数转换为下面的形式（n==3时）：
    // 0 1 a b c
    // 0 0 1 d e
    // 0 0 0 1 f
    //mat[r][c] 表示第r个多项式，第c个系数
    int rank=0; //主元行
    for(int c=0;c<n;c++){
        int pivot=rank;
        for(int r=rank;r<n;r++)
            if(abs(mat[r][c])>abs(mat[pivot][c]))
                pivot=r;
        if(abs(mat[pivot][c])<EPS) //若整列为0，跳过
            continue;
        swap(mat[rank],mat[pivot]); //交换，mat[rank][c]现在是c列中最大的
        double pivotv=mat[rank][c];
        for(int cc=c;cc<=n;cc++) //归一化，使得mat[rank][c]:=1 矩阵同行元素也按比放缩
            mat[rank][cc] /= pivotv;
        for(int r=rank+1;r<n;r++){ //对于剩余行，我们要消掉该未知项的系数。形式地，使别列该项不为0的多项式减去一定的主元行
            if(abs(mat[r][c])>EPS){ //有值
                ld factor=mat[r][c];
                for(int cc=c;cc<=n;cc++) //多项式减法
                    mat[r][cc] -= factor*mat[rank][cc];
            }
        }
        rank++; //该下一个系数
    }
```

```cpp
    bool isno=false,isinf=false; //无解，无穷解。可以得知rank!=n时一定为其中之一
    for(int r=rank;r<n;r++){
        if(abs(mat[r][n])>EPS){ //高斯消元后，如果剩了 0*a_0 + 0*a_1 +...+ 0*a_n = b (b!=0)这种多项式，则无解
            isno=true;
            break;
        }
    }
    if(rank!=n && !isno) //否则，是独立方程少于未知数数量的情况，有无穷个解
        isinf=true;
    if(isno){
        cout<<-1<<'\n';
        return;
    }
    if(isinf){
        cout<<0<<'\n';
        return;
    }
    //回代求解
    //还记得我们想要的形式吗？我们从最下面一行算起即可，这样对于第r行，第r+1~n行的未知项的值都是已知的了
    vector<double> ans(n);
    for(int r=n-1;r>=0;r--){
        ans[r]=mat[r][n];
        for(int c=r+1;c<n;c++){
            ans[r] -= ans[c] * mat[r][c];
        }
    }
    cout<<fixed<<setprecision(2);
    for(int i=0;i<n;i++){
        cout<<'x'<<i<<'='<<ans[i]<<'\n';
    }
}
```
