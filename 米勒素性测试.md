[[数论]]

复杂度：$O\left(klog^{3}n\right)$，该模版对$≤2^{64}$的数字100%正确。
```cpp
inl bool is_prime(int n) {
    if (n < 2) return false;
    if (n == 2 || n == 3 || n == 5) return true;
    if (n % 2 == 0 || n % 3 == 0 || n % 5 == 0) return false;
    if (n < 49) return true; 
    vi bases;
    if (n < 4759123141LL) {
        bases = {2, 7, 61};
    } else if (n < 341550071728321LL) {
        bases = {2, 3, 5, 7, 11, 13, 17};
    } else {
        // n < 2^64
        bases = {2, 325, 9375, 28178, 450775, 9780504, 1795265022};
    }
    // 将 n-1 写成 d * 2^s
    int d = n - 1;
    int s = 0;
    while (d % 2 == 0) {
        d /= 2; s++;
    }
    for (int a : bases) {
        if (a % n == 0) continue;
        int x = qpow(a, d, n); // qpow需使用i128且开头加a%=mod
        if (x == 1 || x == n - 1) continue;
        bool composite = true;
        for (int i = 1; i < s; i++) {
            x = qpow(x, 2, n);
            if (x == n - 1) {
                composite = false; break;
            }
        }
        if (composite) return false;
    }
    return true;
}
```
