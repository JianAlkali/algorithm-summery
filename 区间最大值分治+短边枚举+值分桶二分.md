[[数据结构]]

将值分桶（作为键），将位置作为值，可实现$O\left( n\right)$查询$\left(i,j\right)$对满足条件$a_{i}⊗a_{j}⊕a_{k}=some, 是最任意的数在in\left[i,j\right]上$ 的出现次数。总复杂度为$O\left(n⋅ n\right)$。此处例子为计数所有$\left(i,j\right)$对，满足$a_{i}+a_{j}≡a_{k},a_{k}≔max_{x=\left[i,j\right]}\left(a_{x}\right)$的对数。
```cpp
// 笛卡尔树
struct Car{
    int val[N], ls[N], rs[N];
    int ans;
    // map<int,int> cnt[N];
    unordered_map<int,vi> pos; // val->pos 值的桶，二分值的位置
    int n;
    // 查询 区间最值为uval，当前枚举值为rval与[l,r]上值组合 O(logn)
    // i<j && a_u=[i,j]上的a_ma
    void incal(int uval,int l,int r,int rval){
        int lval=uval-rval; // 此处约束 a_l + a_r = a_u
        if(!pos.count(lval)) return;
        auto& ps = pos[lval];
        auto lit = lower_bound(ps.begin(),ps.end(),l);
        auto rit = upper_bound(lit,ps.end(),r);
        ans += distance(lit,rit);
        return;
    }
    void cal(int u,int lb,int rb){
        if(ls[u]) cal(ls[u],lb,u-1);
        if(rs[u]) cal(rs[u],u+1,rb);
        // 短边枚举
        if(u-lb>rb-u){
            // 右边更短 枚举右边
            for(int i=u+1;i<=rb;i++)   
                incal(val[u],lb,u,val[i]);
            incal(val[u],u+1,rb,val[u]);
        }else{
            // 枚举左边
            for(int i=lb;i<u;i++)
                incal(val[u],u,rb,val[i]);
            incal(val[u],lb,u-1,val[u]);
        }
    }
```

```cpp
int build(int a[N], int _n){
        // 笛卡尔树
        n=_n;
        ans=0;
        pos.clear();
        pos.reserve(n);
        for(int i=1;i<=n;i++){ // 1base
            val[i]=a[i];
            ls[i]=rs[i]=0;
            pos[val[i]].emplace_back(i); // 不用再sort
        }
        stack<int> s;
        for(int i=1;i<=n;i++){
            int p=0;
            while(!s.empty() && val[i]>val[s.top()]){
                int q=s.top(); s.pop();
                rs[q]=p;
                p=q;
            }
            ls[i]=p;
            s.emplace(i);
        }
        int p=0;
        while(!s.empty()){
            int q=s.top(); s.pop();
            rs[q]=p;
            p=q;
        }
        cal(p,1,n);
        return ans;
    }
} car;
```
