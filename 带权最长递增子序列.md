[[动态规划]]

复杂度$O\left(nlogn\right)$，最长递增子序列（LIS）做法暂时不写，其相当于权值全为1的带权最长递增子序列。以下为带权最长不递减子序列（带权LNDS）写法，可按注释扩展。本质上是树状数组+一些维护。
```cpp
#define low(x) ((x)&(-(x)))
struct Fen{ // 前缀最大值树状数组
    int b[N];
    int n;
    inl void update(int p, int v){ // b[p]=v
        while(p<=n){
            b[p]=max(b[p],v);
            p+=low(p);
        }
    }
    inl int query(int r){ // 返回b在[1,r]上的最大值
        int res=0;
        while(r){
            res=max(res,b[r]);
            r-=low(r);
        }
        return res;
    }
    inl void rebuild(int _n){
        n=_n;
        fill(b,b+n+1,0);
    }
}fen;
int a[N],li[N],c[N]; // c为价值
inl void solve(){
    int n;
    cin>>n;
for(int i=1;i<=n;i++)
cin>>a[i];
for(int i=1;i<=n;i++)
cin>>c[i];
    copy(a+1,a+n+1,li); // 使用值域相关的树状数组需离散化
    sort(li,li+n);
    int lisz=unique(li,li+n)-li;
    fen.rebuild(lisz);
```
