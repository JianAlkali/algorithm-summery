[[_数论]]

这是一套矩阵运算的模版。加法$O\left(n^{2}\right)$，乘法$O\left(n^{3}\right)$，快速幂$O\left(n^{3}⋅\log k\right)$。
计算斐波那契数列可用矩阵快速幂，其初始矩阵=`{{1,1},{1,0}}`
```cpp
using Mat=array<array<int,M>,M>;
Mat add(Mat& a,Mat& b,int m=M){
    Mat r={};
    for(int i=0;i<m;i++)
        for(int j=0;j<m;j++)
            r[i][j]=a[i][j]+b[i][j];
    return r;
}
Mat mul(Mat& a,Mat& b,int m=M){
    Mat r={};
    for(int k=0;k<m;k++){
        for(int i=0;i<m;i++){
            int tmp=a[i][k]; // 缓存优化
            for(int j=0;j<m;j++){
                r[i][j]+=tmp*b[k][j];
            }
        }
    }
    return r;
}
Mat qpow(const Mat& a,int b,int m=M){
    Mat r={}, tmp=a;
    for(int i=0;i<m;i++)
        r[i][i]=1; //单位矩阵
    while(true){
        if(b&1) r=mul(r,tmp,m);
        b>>=1;
        if(!b) break;
        tmp=mul(tmp,tmp,m); // 放在这可优化，少乘一次
    }
    return r;
}
```
