[[_图论]]

$O\left(n+m\right)$，有向图中按入边入度排序，在有向无环图上可dp遍历每一个点。或可判环的存在与否。
```cpp
struct E{
    int v,nxt;
}e[N<<1];
int head[N], in[N], ec=0;
void add(int u,int v){
    e[++ec]={v,head[u]};
    head[u]=ec;
    in[v]++; // 统计入边
}
void solve(){
    // 输入部分略
    queue<int> q;
    for(int u=1;u<=n;u++){
        if(in[u]==0) q.emplace(u);
    }
    int cnt=0;
    while(!q.empty()){
        int u=q.front(); q.pop();
        cnt++;
        for(int i=head[u];i;i=e[i].nxt){
            int v=e[i].v;
            // 进行处理如 dp[v]=max(dp[v],dp[u]+a[u]+1);
            if(--in[v]==0){
                q.emplace(v);
            }
        }
    }
    if(cnt!=n){
        cout<<"-1\n"; return; // 有环
    }
}
```
