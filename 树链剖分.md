[[_数据结构]]

复杂度$O\left(N⋅\log N+Q⋅\log ^{2}N\right)$。这是一个构建$O\left(N*\log N\right)$，对树$u →v$路径上（包含`u` `v`）所有点操作$O\left(\log ^{2}N\right)$，对`u`作为根的子树上（包含`u`）所有节点操作$O\left(\log N\right)$的算法。这个模版支持操作：节点带初始值的情况、区间乘法、区间加法、区间赋值、区间求和、$O\left(\log N\right)$大材小用的LCA计算。感谢lxz的线段树模版。线段树采用lxz版本（区间加乘赋值懒标记+取模），此处假设你已经写好了线段树`struct Seg{…}seg;`
```cpp
//树链剖分 Heavy Light Decomposition 重轻分解!!
class HLD{
    int n;
    struct E{
        int v,nxt;
    }edge[N>>1]; //双向边
    int ecnt=1; //1based 边 链式前向星尾巴
    array<int,N> head; //链式前向星头
    array<int,N> fa; //爸爸（字面意）
    array<int,N> top; //zhong重链顶（沿着重边一直走，直到走到没重边可走的那个顶点，这点就是top存的）
    array<int,N> son; //zhong重儿子（多个儿子中任一一个sz最大的）
    array<int,N> sz; //子树大小
    array<int,N> dep; //深度
    array<int,N> idx; //dfs序（用于和线段树交互）
    int nowtm=1; //1based，Seg，配合idx排dfs序
    Seg seg; //线段树 1based 它的构造在HLD的构造部分
    //第一次dfs，计算子树大小、深度、爹、重儿子
    void dfs1(int u,int fath){
        sz[u]=1; //它本身大小也算
        dep[u]=dep[fath]+1; fa[u]=fath;
        int masz=0; //最重儿子的重量
        son[u]=0; //最重儿（默认没有）
        for(int eg=head[u];eg;eg=edge[eg].nxt){
            int v=edge[eg].v;
            if(v==fath) continue;
            dfs1(v,u);
            sz[u]+=sz[v];
            if(sz[v]>masz){ //取一个最重儿
                masz=sz[v]; son[u]=v;
            }
        }
    }

    //第二次dfs，目的就是将树上节点按一种方式排序，来做到：对任一一个操作，调用线段树的次数最多最坏为Ologn
    //这种排序，是：
    /*
    1、其中一种dfs序。dfs序是dfs一遍树，树上的节点被访问的顺序
    2、一个节点的孩子，一定是重儿子排前/小。
    上面这两点，用处如下：
    对一个节点子树做操作：dfs序，在一颗子树上一定连续，而且这颗子树的大小已知，就可以向线段树发出1条区间u~u+sz指令代替，Ologn
    对一个节点即其到根所有节点操作：我们可以证明每次有重链走重链，否则走轻边，一共最多需要跳Ologn次（Ologn个分断的区间指令），证明：
        因为我们每次总是将重儿子作为dfs序最先出现的，那么一条重链的dfs序也都是连续的，那么对一条重链的操作只需要发出一个指令给线段树
        u是v爸爸，u-v是一条轻边，一定有sz[u]>sz[v]*2（否则，该边会成为重边），这意味着根到任一节点的轻边数一定<=logn个(sz倍增)
        u是v重链顶，因为重链之间一定有轻边（否则，其应该会合并为一条重边），而根到一点轻边数量最坏为logn，则重边数量最坏也logn+1
        那么，最坏情况下，我们进行对一个节点到根节点所有节点操作，要对线段树发出Ologn个操作，线段树本身复杂度Ologn，整体复杂度Oqlog^2n
        对任两个节点，也是类似的，对它们的到共同祖先的节点都操作一遍，整体复杂度为Oq*log^2(n)
    */
    void dfs2(int u,int tp){ //tp是top，是重链顶
        idx[u]=nowtm++;
        top[u]=tp;
        if(son[u]){
            dfs2(son[u],tp); //重儿子优先，保证重链上dfs序连续
        }
        for(int eg=head[u];eg;eg=edge[eg].nxt){
            int v=edge[eg].v;
            if(v==fa[u] || v==son[u])
                continue;
            dfs2(v,v); //轻儿子开新链
        }
    }

public:
    HLD(int n):seg(n),n(n){ //:seg(n) 可看做把Seg seg(n)放下面，（但':'操作才能调用类的构造函数）
        fill(all(head,n+1),0);
        fill(all(fa,n+1),0);
        fill(all(son,n+1),0);
        fill(all(sz,n+1),0);
        dep[0]=0;
    }
    HLD(const array<int,N>& a,int n):seg(a,n),n(n){ //带初始节点值的初始化
        fill(all(head,n+1),0);
        fill(all(fa,n+1),0);
        fill(all(son,n+1),0);
        fill(all(sz,n+1),0);
        dep[0]=0;
    }
    void add_edge(int u,int v){ //添加边，添加边自带添加双向边的操作，不要调用两次add_edge
        edge[ecnt]={v,head[u]};
        head[u]=ecnt++;
        edge[ecnt]={u,head[v]};
        head[v]=ecnt++; //反向边
    }
    void build(){ //请在调用完所有add后调用
        dfs1(1,0); //1为根
        dfs2(1,1);
    }
    int query(int u,int v){ //u v之间值查询
        int ans=0;
        while(top[u]!=top[v]){ //重链顶不一样
            if(dep[top[u]]<dep[top[v]]) swap(u,v); //每次选择top最深的节点操作，确保跳到top后不会跳过
            ans+=seg.query(idx[top[u]],idx[u]); //对top[u]~u执行操作，注意dfs序越祖先越小
            u=fa[top[u]]; //top[u]~u已经操作完了，fa[top[u]]是最后一个未处理节点
        }
        if(dep[u]>dep[v]) swap(u,v); //现在u、v重链顶是一样的，它们两个之间的dfs序一定连续了
        ans+=seg.query(idx[u],idx[v]); //那么，我们只需要对它们两个之间的节点操作，注意深度小的节点idx小
        return ans;
    }

    void add(int u,int v,int val){ //对u、v间的点执行相加操作
        while(top[u]!=top[v]){ //同上，这些操作只是对线段树seg调用的函数不同
            if(dep[top[u]]<dep[top[v]]) swap(u,v);
            seg.add(idx[top[u]],idx[u],val);
            u=fa[top[u]];
        }
        if(dep[u]>dep[v]) swap(u,v);
        seg.add(idx[u],idx[v],val);
    }
    void mul(int u,int v,int val){ //同上
        while(top[u]!=top[v]){
            if(dep[top[u]]<dep[top[v]]) swap(u,v);
            seg.mul(idx[top[u]],idx[u],val);
            u=fa[top[u]];
        }
        if(dep[u]>dep[v]) swap(u,v);
        seg.mul(idx[u],idx[v],val);
    }
    void set(int u,int v,int val){ //同上
        while(top[u]!=top[v]){
            if(dep[top[u]]<dep[top[v]]) swap(u,v);
            seg.set(idx[top[u]],idx[u],val);
            u=fa[top[u]];
        }
        if(dep[u]>dep[v]) swap(u,v);
        seg.set(idx[u],idx[v],val);
    }
    int query(int u){ //下面是对子树操作，u是子树根节点。可以知道u~u+sz即可代表u的子树的dfs序的区间
        return seg.query(idx[u],idx[u]+sz[u]-1); //注意减一
    }
    void add(int u,int val){
        seg.add(idx[u],idx[u]+sz[u]-1,val);
    }
    void mul(int u,int val){
        seg.mul(idx[u],idx[u]+sz[u]-1,val);
    }
    void set(int u,int val){
        seg.set(idx[u],idx[u]+sz[u]-1,val);
    }

    int lca(int u,int v){ //她也可以lca哦
        while(top[u]!=top[v]){ //跳法一样，但只需要跳，不用计算
            if(dep[top[u]]<dep[top[v]]) swap(u,v);
            u=fa[top[u]];
        }
        if(dep[u]>dep[v]) swap(u,v); //现在它们top一样，在同一条重链上，dep小的即是lca
        return u;
    }
    void printall(){ //仅作调试输出每个节点的值
        for(int i=1;i<=n;i++){
            cerr<<"["<<i<<"] idx:"<<idx[i]<<" fa:"<<fa[i]<<" top:"<<top[i]<<" son:"<<son[i]<<" sz:"<<sz[i]<<" dep:"<<dep[i];
            cerr<<"  idx["<<i<<"]="<<idx[i]<<" query it:"<<seg.query(idx[i],idx[i])<<'\n';
        }
    }
};
```
