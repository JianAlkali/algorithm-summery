$O(n\log n+q\log n)$实现给定`n`个区间`[l,r]`，`q`个询问`[ql,qr]`，对每个询问，输出`[ql,qr]`中最多能选取的两两不重叠`[l,r]`区间的数量。
```cpp
pii a[N], qs[N];
int li[N];
int nxt[24][N];
void solve(){
    int n;
    cin>>n;
    int lisz=0;
    for(int i=0;i<n;i++)
        cin>>a[i].first>>a[i].second, li[++lisz]=a[i].first, li[++lisz]=a[i].second;
    int m;
    cin>>m;
    for(int i=0;i<m;i++)
        cin>>qs[i].first>>qs[i].second, li[++lisz]=qs[i].first, li[++lisz]=qs[i].second;

    // 1based离散化
    sort(li+1,li+lisz+1);
    lisz = unique(li+1,li+lisz+1) - (li+1);
#define to_li(x) x = lower_bound(li+1,li+lisz+1,x)-li
    for(int i=0;i<n;i++){
        to_li(a[i].first); to_li(a[i].second);
    }
    for(int i=0;i<m;i++){
        to_li(qs[i].first); to_li(qs[i].second);
    }

    // 贪心（右端点最优）->左端点最佳匹配 仅排序O(nlogn)
    sort(a,a+n,[](const pii& l,const pii& r){
        return l.second!=r.second?l.second<r.second : l.first>r.first;
    });
    fill(nxt[0],nxt[0]+lisz+1,INF);
    nxt[0][0]=INF-1;
    for(int i=0;i<n;i++){
        auto [l,r]=a[i];
        while(nxt[0][l]==INF){
            nxt[0][l]=r;
            l--;
        }
    }

    // 倍增跳
    const int O=__lg(n)+1;
    for(int o=1;o<O;o++){
        for(int i=1;i<=lisz;i++){
            int jmp=nxt[o-1][i];
            if(jmp!=INF) nxt[o][i]=nxt[o-1][jmp];
            else nxt[o][i]=INF;
        }
    }

    // query
    for(int q=0;q<m;q++){
        int ans=0;
        auto [ql,qr]=qs[q];
        for(int o=O-1;o>=0;o--){
            if(nxt[o][ql]<=qr){
                ql=nxt[o][ql];
                ans+=1<<o;
            }
        }
        cout<<ans<<'\n';
    }
}
```
